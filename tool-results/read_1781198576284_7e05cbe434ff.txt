   585→function subscribeNewsletter() {
   586→  const email = document.getElementById('newsletterEmail').value.trim();
   587→  
   588→  if (!email) {
   589→    showToast('⚠️ Veuillez entrer votre adresse email', 'warning');
   590→    return;
   591→  }
   592→  
   593→  if (!/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email)) {
   594→    showToast('⚠️ Adresse email invalide', 'warning');
   595→    return;
   596→  }
   597→  
   598→  const subscriptions = JSON.parse(localStorage.getItem('newsletter_subscriptions') || '[]');
   599→  if (subscriptions.find(s => s.email === email)) {
   600→    showToast('⚠️ Cet email est déjà abonné', 'warning');

   602→  }

   874→function startCounterAnimation() {
   875→  const counters = document.querySelectorAll('.stat-number');
   876→  
   877→  const runCounters = () => {
   878→    counters.forEach(counter => {
   879→      const target = parseInt(counter.dataset.target);
   880→      const increment = target / 50;
   881→      let current = 0;
   882→      
   883→      const updateCounter = () => {
   884→        current += increment;
   885→        if (current < target) {
   886→          counter.textContent = Math.ceil(current);
   887→          requestAnimationFrame(updateCounter);
   888→        } else {
   889→          counter.textContent = target + (target >= 10 ? '+' : '');
   890→        }
   891→      };
   892→      
   893→      const observer = new IntersectionObserver((entries) => {
   894→        if (entries[0].isIntersecting && !counter.classList.contains('animated')) {
   895→          counter.classList.add('animated');
   896→          updateCounter();
   897→          observer.unobserve(counter);
   898→        }
   899→      }, { threshold: 0.5 });
   900→      
   901→      observer.observe(counter);
   902→    });
   903→  };
   904→  
   905→  runCounters();
   906→}

   967→async function subscribePushNotifications() {
   968→  if (!('Notification' in window) || !('serviceWorker' in navigator)) {
   969→    showToast('⚠️ Notifications non supportées par ce navigateur', 'warning');
   970→    return false;
   971→  }
   972→  try {
   973→    const permission = await Notification.requestPermission();
   974→    if (permission !== 'granted') {
   975→      showToast('⚠️ Permission de notification refusée', 'warning');
   976→      return false;
   977→    }
   978→    const registration = await navigator.serviceWorker.ready;
   979→    const subscription = await registration.pushManager.subscribe({
   980→      userVisibleOnly: true
   981→    });
   982→    localStorage.setItem('push_subscription', JSON.stringify(subscription));
   983→    showToast('✅ Notifications activées avec succès', 'success');
   984→    return true;
   985→  } catch (err) {
   986→    console.error('Push subscription failed:', err);
   987→    showToast('❌ Échec d\'activation des notifications', 'warning');
   988→    return false;
   989→  }
   990→}

   994→    const registration = await navigator.serviceWorker.ready;
   995→    const subscription = await registration.pushManager.getSubscription();
   996→    if (subscription) {
   997→      await subscription.unsubscribe();
   998→      localStorage.removeItem('push_subscription');
   999→      showToast('✅ Notifications désactivées', 'success');
  1000→    }
  1001→  } catch (err) {
  1002→    console.error('Unsubscribe failed:', err);
  1003→  }
  1004→}

  1094→      data-index="${i}" onmouseenter="highlightSearchItem(${i})">

  1105→function highlightMatch(text, query) {
  1106→  if (!query) return text;
  1107→  const idx = text.toLowerCase().indexOf(query);
  1108→  if (idx === -1) return text;
  1109→  return text.slice(0, idx) + '<strong class="text-cyan-400">' + text.slice(idx, idx + query.length) + '</strong>' + text.slice(idx + query.length);
  1110→}

  1134→    if (e.key === 'ArrowUp') {

  1136→      searchSelectedIndex = Math.max(searchSelectedIndex - 1, 0);
  1137→      items.forEach((el, i) => el.classList.toggle('search-highlight', i === searchSelectedIndex));
  1138→      items[searchSelectedIndex]?.scrollIntoView({ block: 'nearest' });
  1139→    }
  1140→    if (e.key === 'Enter') { e.preventDefault(); navigateSearchResult(searchSelectedIndex); }
  1141→  });
  1142→}
