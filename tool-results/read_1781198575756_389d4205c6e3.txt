   216→function submitTestimonial(e) {

   222→  const testimonial = { text, author: name, role, rating: selectedRating, initials: name.split(' ').map(w => w[0]).join('').substring(0,2).toUpperCase(), color: selectedRating >= 4 ? 'from-cyan-500 to-cyan-600' : 'from-amber-500 to-amber-600', date: new Date().toISOString() };

   227→  document.getElementById('testimonialSuccess').classList.remove('hidden');
   228→  setTimeout(() => {

   567→function handleContact(e) {

   611→    const mailtoLink = `mailto:atia.abdelkader@gmail.com?subject=${encodeURIComponent(subject)}&body=${encodeURIComponent(`الاسم: ${name}\nالبريد: ${email}\n\n${message}`)}`;
   612→    window.location.href = mailtoLink;

   616→    showToast('📧 جاري فتح تطبيق البريد...', 'success');

   620→    submitBtn.innerHTML = originalBtnText;
   621→  });

   640→ * Handle newsletter subscription
   641→ */
   642→function subscribeNewsletter() {
   643→  const email = document.getElementById('newsletterEmail').value.trim();
   644→  
   645→  if (!email) {
   646→    showToast('⚠️ الرجاء إدخال بريدك الإلكتروني', 'warning');
   647→    return;
   648→  }
   649→  
   650→  if (!/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email)) {
   651→    showToast('⚠️ البريد الإلكتروني غير صحيح', 'warning');
   652→    return;
   653→  }
   654→  
   655→  // Save subscription to localStorage
   656→  const subscriptions = JSON.parse(localStorage.getItem('newsletter_subscriptions') || '[]');
   657→  if (subscriptions.find(s => s.email === email)) {
   658→    showToast('⚠️ هذا البريد مسجل مسبقاً في النشرة', 'warning');
   659→    return;
   660→  }
   661→  subscriptions.push({ email, subscribedAt: new Date().toISOString() });
   662→  localStorage.setItem('newsletter_subscriptions', JSON.stringify(subscriptions));
   663→  localStorage.setItem('newsletter_count', subscriptions.length);
   664→  
   665→  showToast('✅ شكرًا! تم اشتراكك بنجاح في النشرة البريدية', 'success');
   666→  document.getElementById('newsletterEmail').value = '';
   667→  if (window.analyticsTracker) {
   668→    window.analyticsTracker.trackEvent('newsletter_subscribe', { email });
   669→  }
   670→}

  1235→/**
  1236→ * Animated counter for statistics
  1237→ */
  1238→function startCounterAnimation() {
  1239→  const counters = document.querySelectorAll('.stat-number');
  1240→  
  1241→  const runCounters = () => {
  1242→    counters.forEach(counter => {
  1243→      const target = parseInt(counter.dataset.target);
  1244→      const increment = target / 50; // Animate in 50 steps
  1245→      let current = 0;
  1246→      
  1247→      const updateCounter = () => {
  1248→        current += increment;
  1249→        if (current < target) {
  1250→          counter.textContent = Math.ceil(current);
  1251→          requestAnimationFrame(updateCounter);
  1252→        } else {
  1253→          counter.textContent = target + (target >= 10 ? '+' : '');
  1254→        }
  1255→      };
  1256→      
  1257→      // Trigger animation when element is in view
  1258→      const observer = new IntersectionObserver((entries) => {
  1259→        if (entries[0].isIntersecting && !counter.classList.contains('animated')) {
  1260→          counter.classList.add('animated');
  1261→          updateCounter();
  1262→          observer.unobserve(counter);
  1263→        }
  1264→      }, { threshold: 0.5 });
  1265→      
  1266→      observer.observe(counter);
  1267→    });
  1268→  };
  1269→  
  1270→  runCounters();
  1271→}

  1358→  } catch (err) {
  1359→    console.error('Push subscription failed:', err);

  1361→    return false;
  1362→  }
