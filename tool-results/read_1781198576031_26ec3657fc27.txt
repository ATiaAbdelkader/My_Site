   217→function submitTestimonial(e) {

   219→  const name = document.getElementById('tName').value.trim();
   220→  const role = document.getElementById('tRole').value.trim();
   221→  const text = document.getElementById('tText').value.trim();

   223→  const testimonial = { text, author: name, role, rating: selectedRating, initials: name.split(' ').map(w => w[0]).join('').substring(0,2).toUpperCase(), color: selectedRating >= 4 ? 'from-cyan-500 to-cyan-600' : 'from-amber-500 to-amber-600', date: new Date().toISOString() };

   225→  saved.push(testimonial);
   226→  localStorage.setItem('user_testimonials', JSON.stringify(saved));
   227→  document.getElementById('testimonialForm').reset();
   228→  document.getElementById('testimonialSuccess').classList.remove('hidden');
   229→  // Re-render testimonials to include new one
   230→  setTimeout(() => {

   232→    if (typeof renderTestimonials === 'function') renderTestimonials();

   236→    method: 'POST', headers: { 'Accept': 'application/json', 'Content-Type': 'application/json' },
   237→    body: JSON.stringify({ name, role, text, rating: selectedRating, type: 'testimonial' })

   572→// Contact form
   573→/**
   574→ * Handle contact form submission
   575→ * @param {Event} e - Form submit event
   576→ */
   577→function handleContact(e) {
   578→  e.preventDefault();
   579→  const form = document.getElementById('contactForm');
   580→  const name = document.getElementById('name').value.trim();
   581→  const email = document.getElementById('email').value.trim();
   582→  const subject = document.getElementById('subject').value;
   583→  const message = document.getElementById('message').value.trim();
   584→  const success = document.getElementById('formSuccess');
   585→  const submitBtn = form.querySelector('button[type="submit"]');
   586→  const originalBtnText = submitBtn.innerHTML;
   587→  
   588→  // Validation
   589→  if (!name || !email || !subject || !message) {
   590→    showToast('⚠️ Please fill in all required fields', 'warning');
   591→    return;
   592→  }
   593→  
   594→  if (!/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email)) {
   595→    showToast('⚠️ Invalid email address', 'warning');
   596→    return;
   597→  }
   598→  
   599→  // Show loading state
   600→  submitBtn.disabled = true;
   601→  submitBtn.innerHTML = '<span>Sending...</span>';
   602→  
   603→  // Send via Formspree with mailto fallback
   604→  const formspreeId = localStorage.getItem('formspree_id') || 'ada15_agro@';
   605→  
   606→  fetch(`https://formspree.io/f/${formspreeId}`, {
   607→    method: 'POST',
   608→    body: JSON.stringify({ name, email, subject, message }),
   609→    headers: { 'Content-Type': 'application/json', 'Accept': 'application/json' }
   610→  })
   611→  .then(response => {
   612→    if (response.ok) {
   613→      success.classList.remove('hidden');
   614→      form.reset();
   615→      setTimeout(() => success.classList.add('hidden'), 4000);
   616→      showToast('✅ Message sent successfully! I\'ll be in touch soon.', 'success');
   617→    } else {
   618→      throw new Error('Formspree error');
   619→    }
   620→  })
   621→  .catch(() => {
   622→    // Fallback to mailto:
   623→    const mailtoLink = `mailto:atia.abdelkader@gmail.com?subject=${encodeURIComponent(subject)}&body=${encodeURIComponent(`Name: ${name}\nEmail: ${email}\n\n${message}`)}`;
   624→    window.location.href = mailtoLink;
   625→    success.classList.remove('hidden');
   626→    form.reset();
   627→    setTimeout(() => success.classList.add('hidden'), 4000);
   628→    showToast('📧 Opening your email client...', 'success');
   629→  })
   630→  .finally(() => {
   631→    submitBtn.disabled = false;
   632→    submitBtn.innerHTML = originalBtnText;
   633→  });
   634→}

   677→  showToast('✅ Thank you! You\'ve subscribed successfully', 'success');
   678→  document.getElementById('newsletterEmail').value = '';

   680→    window.analyticsTracker.trackEvent('newsletter_subscribe', { email });

   965→/**
   966→ * Animated counter for statistics
   967→ */
   968→function startCounterAnimation() {
   969→  const counters = document.querySelectorAll('.stat-number');
   970→  
   971→  const runCounters = () => {
   972→    counters.forEach(counter => {
   973→      const target = parseInt(counter.dataset.target);
   974→      const increment = target / 50; // Animate in 50 steps
   975→      let current = 0;
   976→      
   977→      const updateCounter = () => {
   978→        current += increment;
   979→        if (current < target) {
   980→          counter.textContent = Math.ceil(current);
   981→          requestAnimationFrame(updateCounter);
   982→        } else {
   983→          counter.textContent = target + (target >= 10 ? '+' : '');
   984→        }
   985→      };
   986→      
   987→      // Trigger animation when element is in view
   988→      const observer = new IntersectionObserver((entries) => {
   989→        if (entries[0].isIntersecting && !counter.classList.contains('animated')) {
   990→          counter.classList.add('animated');
   991→          updateCounter();
   992→          observer.unobserve(counter);
   993→        }
   994→      }, { threshold: 0.5 });
   995→      
   996→      observer.observe(counter);
   997→    });
   998→  };
   999→  
  1000→  runCounters();
  1001→}

  1069→// Push notification subscription
  1070→async function subscribePushNotifications() {
  1071→  if (!('Notification' in window) || !('serviceWorker' in navigator)) {
  1072→    showToast('⚠️ Notifications not supported in your browser', 'warning');
  1073→    return false;
  1074→  }
  1075→  try {
  1076→    const permission = await Notification.requestPermission();
  1077→    if (permission !== 'granted') {
  1078→      showToast('⚠️ Notification permission denied', 'warning');
  1079→      return false;
  1080→    }
  1081→    const registration = await navigator.serviceWorker.ready;
  1082→    const subscription = await registration.pushManager.subscribe({
  1083→      userVisibleOnly: true
  1084→    });
  1085→    localStorage.setItem('push_subscription', JSON.stringify(subscription));
  1086→    showToast('✅ Notifications enabled successfully', 'success');
  1087→    return true;
  1088→  } catch (err) {
  1089→    console.error('Push subscription failed:', err);
  1090→    showToast('❌ Failed to enable notifications', 'warning');
  1091→    return false;
  1092→  }
  1093→}

  1095→async function unsubscribePushNotifications() {
  1096→  try {
  1097→    const registration = await navigator.serviceWorker.ready;
  1098→    const subscription = await registration.pushManager.getSubscription();
  1099→    if (subscription) {
  1100→      await subscription.unsubscribe();
  1101→      localStorage.removeItem('push_subscription');
  1102→      showToast('✅ Notifications disabled', 'success');
  1103→    }
  1104→  } catch (err) {
  1105→    console.error('Unsubscribe failed:', err);
  1106→  }
  1107→}

  1225→function initSearch() {
  1226→  buildSearchIndex();
  1227→  const input = document.getElementById('searchInput');
  1228→  if (!input) return;
  1229→  input.addEventListener('input', (e) => renderSearchResults(e.target.value));
  1230→  input.addEventListener('keydown', (e) => {
  1231→    const items = document.querySelectorAll('.search-result-item');
  1232→    if (e.key === 'ArrowDown') {
  1233→      e.preventDefault();
  1234→      searchSelectedIndex = Math.min(searchSelectedIndex + 1, items.length - 1);
  1235→      items.forEach((el, i) => el.classList.toggle('search-highlight', i === searchSelectedIndex));
  1236→      items[searchSelectedIndex]?.scrollIntoView({ block: 'nearest' });
  1237→    }
  1238→    if (e.key === 'ArrowUp') {
  1239→      e.preventDefault();
  1240→      searchSelectedIndex = Math.max(searchSelectedIndex - 1, 0);
